40template <typename key, typename value>
41std::unordered_map<value, key>
42InvertMap(const std::unordered_map<key, value> &originalMap,
43 const std::string &errorOnDuplicateValue)
44{
45 std::unordered_map<value, key> result;
46 for (const auto &[orig_key, orig_value] : originalMap)
47 {
48 auto success = result.emplace(orig_value, orig_key);
49 if (not success.second)
50 {
51 throw std::runtime_error(errorOnDuplicateValue);
52 }
53 }
54
55 return result;
56}
